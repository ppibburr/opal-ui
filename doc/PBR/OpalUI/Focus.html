def initialize *o
  super
  
  element[:tabindex] = "0"
  on :focus do
    $document.css(".pbr-widget-focus").each do |c|
      c.remove_class "pbr-widget-focus"
    end
    element.add_class "pbr-widget-focus"          
  end
end
