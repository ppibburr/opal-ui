def activate
  if !focused?
    focus()
  end
  
  element.add_class "pbr-opalui-active-activated"    
  
  @on_activate_cb.call(self) if @on_activate_cb
  
  return self
end

def initialize *o
  super
  
  element.on :keypress do |e|
    if (e.code == 13)
        activate()
        next false;
    end
      
    next true;
  end
end
