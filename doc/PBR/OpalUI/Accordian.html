def append(opts={}, &b)
  item = self.class::Item.new
  
  if w = opts.delete(:child)
    item.add(w)
  end
  
  item.apply(opts)
  
  append_item item
  
  add item
  
  item.select if children.length == 1
  
  b.call(item) if b

  return self
end
