def add w
  w.element.append_to container_element
  
  bool = false
  
  self.class.ancestors.reverse.each do |q|
    bool = true if q == Container      
    
    next unless bool
    
    w.element.add_class q.class_name+"-child"
  end

  return self
end

def children &b
  a = []
  
  container_element.css(".pbr-opalui-container-child").each do |c|
    w = Widget.wrap c
    yield w if b
    a << w
  end
  
  a
end
