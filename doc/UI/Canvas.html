def initialize(tag="CANVAS", wrap=nil, init=nil)
  super tag,wrap,init
  
  if (wrap and init) or tag
    @canvas = Browser::Canvas.new(element)
  else
    raise "NotImplemented: wrap initialized CANVAS"
  end
  
  @text = @canvas.text
  @style = @canvas.style
end
